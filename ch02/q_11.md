```javascript
// Coded solution to question 11

/**
 * 11. Rewrite all the for loops in Section 2.10, "for Loops" (page 41), as while loops.
 */

// Loop 1
/**
 * for(let i = 1; i <= 10; i++)
 *     console.log(i)
 */
{
    let i = 1;
    while(i <= 10) {
        console.log(i);
        i++;
    }

    console.log();      // to format the output
}

// Loop 2
/**
 * for(let i = a.length - 1; i >= 0; i--)
 *     console.log(a[i])
 */
{
    let a = [1, 2, 3, 4, 5];

    let i = a.length - 1;

    while(i >= 0) {
        console.log(a[i]);
        i--;
    }

    console.log();
}

// Loop 3
/**
 * let arr = [, 2, , 4]
 * arr[9] = 100
 * for(const element of arr)
 *     console.log(element)
 */
{
    let arr = [, 2, , 4];

    arr[9] = 100;

    let i = 0;

    while(i < arr.length) {
        console.log(arr[i]);
        i++;
    }

    console.log();
}

// Loop 4
/**
 * let greeeting = 'Hello ðŸŒ'
 * for(const c of greeting)
 *     console.log(c)
 */
{
    let greeting = 'Hello ðŸŒ';          

    let i = 0;

    while(i < greeting.length) {
        console.log(greeting[i]);       // I am sure there is a way to output 'ðŸŒ' inside a while loop,
        i++;                            // but will assume using a for of loop is more sensible.
    }

    console.log();
}

// Loop 5
/**
 * let obj = { name: 'Harry Smith', age: 42 }
 * for(const key in obj)
 *     console.log(`${key}: ${obj[key]}`)
 */
{
    let obj = { name: 'Harry Smith', age: 42 };

    let keys = Object.keys(obj);

    let i = 0;

    while(i < keys.length) {
        console.log(`${keys[i]}: ${obj[keys[i]]}`);
        i++;
    }

    console.log();
}

// Loop 6
/**
 * let numbers = [1, 2, , 4]
 * numbers[99] = 100
 * for(const i in numbers)
 *     console.log(`${i}: ${numbers[i]}`)
 */
{
    let numbers = [1, 2, , 4];

    numbers[99] = 100;

    let i = 0;

    while(i < numbers.length) {
        if(numbers[i] !== undefined) console.log(`${i}: ${numbers[i]}`);
        i++
    }

    console.log();
}

// Loop 7
/**
 * numbers.lucky = true
 * for(const i in numbers)
 *     console.log(`${i}: ${numbers[i]}`)
 */
{
    let numbers = [1, 2, , 4];

    numbers[99] = 100;

    numbers.lucky = true;

    let i = 0;

    let keys = Object.keys(numbers);

    while(i < keys.length) {
        console.log(`${keys[i]}: ${numbers[keys[i]]}`)
        i++
    }

    console.log();
}
```
